pm.test("Status code is 400 or 500", function () {
    pm.expect([400, 500]).to.include(pm.response.code);
});

pm.test("Error message is one of the expected messages", function () {
    const errorMessages = [
        "Internal Server Error",
        "Bad Request"
    ];
    
    pm.expect(
        errorMessages.some(msg => pm.response.text().includes(msg))
    ).to.be.true;
});
